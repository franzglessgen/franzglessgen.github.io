    def multi_gaussian(x, *params):
    """Three Gaussian peaks: params = [amp1, mu1, sigma1, amp2, mu2, sigma2, amp3, mu3, sigma3]"""
    assert len(params) == 9
    g1 = params[0] * np.exp(-0.5 * ((x - params[1]) / params[2])**2)
    g2 = params[3] * np.exp(-0.5 * ((x - params[4]) / params[5])**2)
    g3 = params[6] * np.exp(-0.5 * ((x - params[7]) / params[8])**2)
    return g1 + g2 + g3

    def peak_finding(rough_counts, rough_bins):

    """

    Look for approximate mean and sigma of gaussian peaks in a histogram 


    """


    derivative = np.diff(rough_counts)


    means = [] 
    sigmas = []
    amplitudes = []

    #Find local maxima:

    #Number of steps before sign inversion
    step_counter = 0 
    sign_ref = 1


    for i_d, d in enumerate(derivative):

        #Goes up 
        if np.sign(d) == sign_ref:
            step_counter+=1

        #Goes down, means we found a peak
        else:

            position = rough_bins[i_d]

            means.append(position)
            sigmas.append(step_counter*(rough_bins[1] - rough_bins[0]))
            amplitudes.append(rough_counts[i_d])
            step_counter = 0
            sign_ref *=-1 

    means = np.array(means)
    sigmas = np.array(sigmas)
    amplitudes = np.array(amplitudes)

    means = means[sigmas>5*(rough_bins[1] - rough_bins[0])]
    amplitudes = amplitudes[sigmas>5*(rough_bins[1] - rough_bins[0])]
    sigmas = sigmas[sigmas>5*(rough_bins[1] - rough_bins[0])]


    return means, sigmas, amplitudes

    def two_gaussian(x, *params):
    """Three Gaussian peaks: params = [amp1, mu1, sigma1, amp2, mu2, sigma2]"""
    assert len(params) == 6
    g1 = params[0] * np.exp(-0.5 * ((x - params[1]) / params[2])**2)
    g2 = params[3] * np.exp(-0.5 * ((x - params[4]) / params[5])**2)
    return g1 + g2

    def load_hits_from_root(self, include_zero = True):


        """

        Remove hits around threshold for Qcalib using the argument include_zero = False

        """



        for k_r, v_r in self.meta_data["Xray_runs"].items():

            for v_p in v_r["path_to_runs"]:

                path_to_file = os.path.join(v_r["path_to_data"], v_p)
                print(path_to_file)
                hit_file = ROOT.TFile.Open(path_to_file)

                chip_name = "rd53b_" + v_r["h_and_c_ID"][0] + v_r["h_and_c_ID"][1]

                tree = hit_file.Get("ClusteringSpatial/"+chip_name+"/PixelHits")

                for entry in tree:
                    pixels_per_cluster = entry.clusterNumPixels
                    tot = entry.pixelToT
                    col = entry.pixelX
                    row = entry.pixelY

                    current_pix = 0

                    #Could load the cluster charge directly, but need pixel by pixel gain conversion
                    for i_np, np in enumerate(pixels_per_cluster):

                        VCAL = 0
                        #p denotes the number of pixels in that cluster, i_p is the cluster index
                        for p in range(np):

                            pix_tot = tot[current_pix + p]
                            pix_col = col[current_pix + p]
                            pix_row = row[current_pix + p]


                            if pix_tot == 0 and not include_zero:
                                continue


                            #VCAL = (ToT - b)/a

    def load_gain_from_root(self):

    def load_hits_from_root(self, include_zero = True):


    """

    Remove hits around threshold for Qcalib using the argument include_zero = False

    """



    for k_r, v_r in self.meta_data["Xray_runs"].items():

        for v_p in v_r["path_to_runs"]:

            path_to_file = os.path.join(v_r["path_to_data"], v_p)
            print(path_to_file)
            hit_file = ROOT.TFile.Open(path_to_file)

            chip_name = "rd53b_" + v_r["h_and_c_ID"][0] + v_r["h_and_c_ID"][1]

            tree = hit_file.Get("ClusteringSpatial/"+chip_name+"/PixelHits")

            for entry in tree:
                pixels_per_cluster = entry.clusterNumPixels
                tot = entry.pixelToT
                col = entry.pixelX
                row = entry.pixelY

                current_pix = 0

                #Could load the cluster charge directly, but need pixel by pixel gain conversion
                for i_np, np in enumerate(pixels_per_cluster):

                    VCAL = 0
                    #p denotes the number of pixels in that cluster, i_p is the cluster index
                    for p in range(np):

                        pix_tot = tot[current_pix + p]
                        pix_col = col[current_pix + p]
                        pix_row = row[current_pix + p]


                        if pix_tot == 0 and not include_zero:
                            continue


                        #VCAL = (ToT - b)/a

                        if self.pixel_gain[v_r["source"]][pix_col, pix_row, 1] == 0:
                            continue

                        increment =  (pix_tot - self.pixel_gain[v_r["source"]][pix_col, pix_row, 0])/self.pixel_gain[v_r["source"]][pix_col, pix_row, 1]

                    current_pix += np
